select d.id, e.name, e.position, a.status
from core_developer d, core_entity e, core_analysis a, core_case c, core_case_entities ce, core_smell s
where s.slug='SS' and a.experiment_id=12 and a.smell_id = s.id and a.developer_id = d.id and a.case_id = c.id and ce.case_id=c.id and ce.entity_id=e.id and a.status <> -1
order by d.id;

select d.id, c.id as case_id, a.status
from core_developer d, core_analysis a, core_case c, core_smell s
where s.slug='SS' and a.experiment_id=12 and a.smell_id = s.id and a.developer_id = d.id and a.case_id = c.id and a.status <> -1
order by d.id, c.id;

select distinct d.id, d.name
from core_developer d, core_analysis a, core_case c, core_smell s
where s.slug='SS' and a.experiment_id=12 and a.smell_id = s.id and a.developer_id = d.id and a.case_id = c.id and a.status <> -1
order by d.id;
